reverse proxy server level

Disable Traefik:
	sudo k3s kubectl delete deploy traefik -n kube-system

sudo apt install -y podman podman-compose

sudo podman pull docker.io/jc21/nginx-proxy-manager:latest
sudo podman pull docker.io/justsky/goaccess-for-nginxproxymanager:latest

docker-compose.yml:

services:
  nginx_proxy_manager:
    image: "jc21/nginx-proxy-manager:latest"
    container_name: nginx_proxy_manager
    restart: unless-stopped
    ports:
      - "80:80"    # HTTP
      - "443:443"  # HTTPS
      - "81:81"    # Admin Panel
    volumes:
      - ./data/npm_data:/data
      - ./data/npm_letsencrypt:/etc/letsencrypt
      - ./data/npm_logs:/var/log/nginx
    environment:
      DB_SQLITE_FILE: "/data/database.sqlite" # Using SQLite instead of MySQL for simplicity
      INITIAL_ADMIN_EMAIL: admin@example.com
      INITIAL_ADMIN_PASSWORD: changeme


  goaccess:
    image: justsky/goaccess-for-nginxproxymanager:latest
    container_name: goaccess
    restart: unless-stopped
    environment:
        - TZ=America/Denver
        #- SKIP_ARCHIVED_LOGS=False #optional
        #- BASIC_AUTH=False #optional
        #- BASIC_AUTH_USERNAME=user #optional
        #- BASIC_AUTH_PASSWORD=pass #optional
    ports:
        - '7880:7880'
    volumes:
        - ./data/npm_data/logs:/opt/log
          # - ./data/npm_data/logs:/opt/log

sudo podman-compose up -d

go ip:81 log in with: 
	usr: admin@example.com
	pass: changeme
